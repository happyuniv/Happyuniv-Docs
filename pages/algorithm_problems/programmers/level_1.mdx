```js
function solution(friends, gifts) {
    var answer = 0;
    
    const table = {};
    
    for(friend of friends) {
        table[friend] = {score:0};
    }
        
    for(gift of gifts) {
        const [giver,taker] = gift.split(' ');
        table[giver].score++;
        table[taker].score--;
        
        if(table[giver][taker]) table[giver][taker]++;
        else table[giver][taker] = 1;
    }
    
    for(let i=0;i<friends.length;i++) {
        let count = 0;
        
        for(let j=0;j<friends.length;j++) {
            if(i===j) continue;
            const gives = table[friends[i]][friends[j]];
            const takes = table[friends[j]][friends[i]];
            const giver_score = table[friends[i]].score;
            const taker_score = table[friends[j]].score;
            
            if(gives) {
                if(takes) {
                    if(gives > takes) count++;
                    else if(gives === takes) {
                        if(giver_score > taker_score) count++;
                    }
                }
                else count++;
            }
            else if(!takes) {
                if(giver_score > taker_score) count++;
            }
        }
        
        if(count > answer) answer = count;
    }
    
    return answer;
}
```

```js
function solution(bandage, health, attacks) {
    var answer = 0;
    
    let current_health = health;
    let prev_t = 0;
    for(const attack of attacks) {
        const [t,damage] = attack;

        const heal = (t-prev_t-1)*bandage[1];
        current_health += Math.floor((t-prev_t-1)/bandage[0])*bandage[2];
        current_health += heal;
        if(current_health > health) current_health = health;
        
        current_health -= damage;
        if(current_health <= 0) return -1;

        prev_t = t;
    }
    answer = current_health;
    return answer;
}
```

```js
      if (park[i][j] === 'S') {
        x = i
        y = j

      }

    const num = parseInt(routes[i][2])

      if (park[x - num]) {
        let obstacle = false
        for (let i = 1; i <= num; i++) {
          if (park[x - i][y] === 'X') {
            obstacle = true
            break
          }
        }
        if (obstacle) continue
        else x -= num
      } else continue
    } else if (dir === 'S') {
      if (park[x + num]) {
        let obstacle = false
        for (let i = 1; i <= num; i++) {
          if (park[x + i][y] === 'X') {
            obstacle = true
            break
          }
        }
        if (obstacle) continue
        else x += num
      } else continue
    } else if (dir === 'W') {
      if (park[x][y - num]) {
        let obstacle = false
        for (let i = 1; i <= num; i++) {
          if (park[x][y - i] === 'X') {
            obstacle = true
            break
          }
        }
        if (obstacle) continue
        else y -= num
      } else continue
    } else if (dir === 'E') {
      if (park[x][y + num]) {
        let obstacle = false
        for (let i = 1; i <= num; i++) {
          if (park[x][y + i] === 'X') {
            obstacle = true
            break
          }
        }
        if (obstacle) continue
        else y += num
      } else continue
    }
  }

  return answer
}
```

```js
function solution(today, terms, privacies) {
  var answer = []

  const todayArr = today.split('.')
  const todayYear = parseInt(todayArr[0])
  const todayMonth = parseInt(todayArr[1])
  const todayDay = parseInt(todayArr[2])

  const termsTable = {}
  for (let i = 0; i < terms.length; i++) {
    const termsArr = terms[i].split(' ')
    termsTable[termsArr[0]] = termsArr[1]
  }

  for (let i = 0; i < privacies.length; i++) {
    const privaciesArr = privacies[i].split(' ')
    const dateArr = privaciesArr[0].split('.')

    let expireYear = parseInt(dateArr[0])
    let expireMonth = parseInt(dateArr[1])
    let expireDay = parseInt(dateArr[2])

    const passMonth = parseInt(termsTable[privaciesArr[1]] % 12)
    const passYear = parseInt(termsTable[privaciesArr[1]] / 12)
    expireMonth += passMonth
    expireYear += passYear

    if (expireMonth > 12) {
      expireYear += 1
      expireMonth -= 12
    }

    expireDay -= 1
    if (expireDay === 0) {
      expireDay = 28
      expireMonth -= 1
      if (expireMonth === 0) {
        expireMonth = 12
        expireYear -= 1
      }
    }

    if (expireYear < todayYear) {
      answer.push(i + 1)
    } else if (expireYear === todayYear) {
      if (expireMonth < todayMonth) {
        answer.push(i + 1)
      } else if (expireMonth === todayMonth) {
        if (expireDay < todayDay) {
          answer.push(i + 1)
        }
      }
    }
  }
  return answer
}
```

```js
function solution(nums) {
    var answer = 0;
    
    nums.reduce((acc,cur) => {
        if(acc[cur]) return acc;
        answer++;
        acc[cur] = true;
        return acc;
    },{})
```

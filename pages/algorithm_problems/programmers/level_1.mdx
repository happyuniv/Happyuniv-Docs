```js
function solution(players, callings) {
  var answer = []
  answer = Array.from(players)

  let temp_players = {}
  for (let i = 0; i < players.length; i++) {
    temp_players[players[i]] = i
  }

  for (let calling of callings) {
    let num = temp_players[calling]
    let temp = answer[num]
    answer[num] = answer[num - 1]
    answer[num - 1] = temp

    temp_players[answer[num]] = num
    temp_players[answer[num - 1]] = num - 1
  }

  return answer
}
```

```js
function solution(name, yearning, photo) {
  var answer = []

  let table = {}
  for (let i = 0; i < name.length; i++) {
    table[name[i]] = yearning[i]
  }

  for (let i = 0; i < photo.length; i++) {
    let score = 0
    for (let j = 0; j < photo[i].length; j++) {
      if (!table[photo[i][j]]) continue
      score += table[photo[i][j]]
    }
    answer.push(score)
  }

  return answer
}
```

## 공원 산책

```js
function solution(park, routes) {
  var answer = []

  let x, y

  for (let i = 0; i < park.length; i++) {
    for (let j = 0; j < park[i].length; j++) {
      if (park[i][j] === 'S') {
        x = i
        y = j
        break
      }
    }
  }

  for (let i = 0; i < routes.length; i++) {
    const dir = routes[i][0]
    const num = parseInt(routes[i][2])

    if (dir === 'N') {
      if (park[x - num]) {
        let obstacle = false
        for (let i = 1; i <= num; i++) {
          if (park[x - i][y] === 'X') {
            obstacle = true
            break
          }
        }
        if (obstacle) continue
        else x -= num
      } else continue
    } else if (dir === 'S') {
      if (park[x + num]) {
        let obstacle = false
        for (let i = 1; i <= num; i++) {
          if (park[x + i][y] === 'X') {
            obstacle = true
            break
          }
        }
        if (obstacle) continue
        else x += num
      } else continue
    } else if (dir === 'W') {
      if (park[x][y - num]) {
        let obstacle = false
        for (let i = 1; i <= num; i++) {
          if (park[x][y - i] === 'X') {
            obstacle = true
            break
          }
        }
        if (obstacle) continue
        else y -= num
      } else continue
    } else if (dir === 'E') {
      if (park[x][y + num]) {
        let obstacle = false
        for (let i = 1; i <= num; i++) {
          if (park[x][y + i] === 'X') {
            obstacle = true
            break
          }
        }
        if (obstacle) continue
        else y += num
      } else continue
    }
  }

  answer.push(x, y)

  return answer
}
```

```js
function solution(keymap, targets) {
    var answer = [];
    
    for(let i=0;i<targets.length;i++) {
        
        let count = 0;
        let found = true;
        for(let j=0;j<targets[i].length;j++) {
            let min=101;
            
            for(let k=0;k<keymap.length;k++) {
                for(let l=0;l<keymap[k].length;l++) {  
                    if(keymap[k][l] === targets[i][j]) {
                        if(l < min) {
                            min = l+1;          
                            break;
                        }
                    }         
                }
            }
            if(min === 101) {
                found = false;
                break;
            }
            else count+=min
        }
        
        if(found) answer.push(count);
        else answer.push(-1)
    }
    return answer;
}
```

```js
function solution(cards1, cards2, goal) {
    var answer = 'Yes';

    let card1_index = 0;
    let card2_index = 0;
    
    for(let i=0; i<goal.length; i++) {
        
        if(cards1[card1_index] !== goal[i] && cards2[card2_index] !== goal[i]) {
            return 'No';
        }
        
        if(cards1[card1_index] === goal[i]) {
            card1_index++;
        }
        if(cards2[card2_index] === goal[i]) {
            card2_index++;
        }
    }
    return answer;
}
```

```js
function solution(s, skip, index) {
    var answer = '';
    
    let checkList = {}
    
    for(let i=0; i<skip.length;i++) {
        checkList[skip[i]] = true
    }
    
    for(let i=0; i<s.length;i++) {
        
        let charCode = s.charCodeAt(i);
        for(let j=index;j>0;j--) {
             
            charCode+=1;
            if(charCode === 123) charCode = 97 
            if(checkList[String.fromCharCode(charCode)]) j++
        }
        
        answer+= String.fromCharCode(charCode)
    }
    return answer;
}
```

## 개인정보 수집 유효기간

https://school.programmers.co.kr/learn/courses/30/lessons/150370

```js
function solution(today, terms, privacies) {
    var answer = [];
    
    const todayArr = today.split('.');
    const todayYear = parseInt(todayArr[0]);
    const todayMonth = parseInt(todayArr[1]);
    const todayDay = parseInt(todayArr[2]);
        
    const termsTable = {};
    for(let i=0; i<terms.length; i++) {
        const termsArr = terms[i].split(' ');
        termsTable[termsArr[0]] = termsArr[1]; 
    }
    
    for(let i=0; i< privacies.length; i++) {
        const privaciesArr = privacies[i].split(' ');
        const dateArr = privaciesArr[0].split('.');
        
        let expireYear = parseInt(dateArr[0]);
        let expireMonth = parseInt(dateArr[1]);
        let expireDay = parseInt(dateArr[2]);
        
        const passMonth = parseInt(termsTable[privaciesArr[1]] % 12);
        const passYear = parseInt(termsTable[privaciesArr[1]] / 12);
        expireMonth += passMonth;
        expireYear += passYear;
        
        
        if(expireMonth >12) {
            expireYear+=1;
            expireMonth-=12;
        }
        
        expireDay -= 1;
        if(expireDay === 0) {
            expireDay = 28
            expireMonth -= 1;
            if(expireMonth === 0) {
                expireMonth = 12;
                expireYear -= 1;
            }
        }
                
        if(expireYear < todayYear) {
            answer.push(i+1);
        }
        else if(expireYear === todayYear) {
            if(expireMonth < todayMonth) {
                answer.push(i+1);
            }
            else if(expireMonth === todayMonth) {
                if(expireDay < todayDay) {
                    answer.push(i+1);
                }
            }
        }
    }
    return answer;
}
```

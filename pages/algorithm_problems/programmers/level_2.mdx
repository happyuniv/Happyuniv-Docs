## 요격 시스템

https://school.programmers.co.kr/learn/courses/30/lessons/181188

```js
function solution(targets) {
    var answer = 0;
    
    targets.sort((a,b) => a[0] - b[0]);
    
    let start = 0;
    let end = 0;
    for(let target of targets) {
        if(target[0] >= end) {
            start = target[0];
            end = target[1];
            answer++;
        }
        else if(target[1] < end) {
            end = target[1];
        }
    }
    
    return answer;
}
```

## 두 원 사이의 정수쌍

https://school.programmers.co.kr/learn/courses/30/lessons/181187

```js
function solution(r1, r2) {
    var answer = 0;
    
    let count1 = 0;
    for(let i=1;i<r1;i++) {
        if(Number.isInteger(Math.sqrt((r1+i)*(r1-i)))) count1--;
        count1 += Math.floor(Math.sqrt((r1+i)*(r1-i)));
    }
    
    let count2 = 0;
    for(let i=1;i<r2;i++) {
        count2 += Math.floor(Math.sqrt((r2+i)*(r2-i)));
    }
    
    answer += 4*(r2-r1+1) + 4*(count2-count1);
    
    return answer;
}
```

## *연속된 부분 수열의 합

https://school.programmers.co.kr/learn/courses/30/lessons/178870

```js
function solution(sequence, k) {
    var answer = [];
    
    let end = 0;
    let sum = 0;
    for(let i=0;i<sequence.length;i++) {
        while(sum < k && end < sequence.length) {
            sum += sequence[end];
            end++;
        }
        
        if(sum === k) answer.push([i,end-1]);
        
        sum -= sequence[i];
    }
    
    answer.sort((a,b) => (a[1]-a[0])-(b[1]-b[0]));
    answer = answer[0];
    
    return answer;
}
```

function solution(players, callings) {

  answer = Array.from(players)

  let temp_players = {}
  for (let i = 0; i < players.length; i++) {
    temp_players[players[i]] = i
  }

  for (let calling of callings) {
    let num = temp_players[calling]
    let temp = answer[num]
    answer[num] = answer[num - 1]
    answer[num - 1] = temp

    temp_players[answer[num]] = num
    temp_players[answer[num - 1]] = num - 1
  }

  return answer
}

function solution(park, routes) {
  var answer = []

  let x, y

  for (let i = 0; i < park.length; i++) {
    for (let j = 0; j < park[i].length; j++) {
      if (park[i][j] === 'S') {
        x = i
        y = j
        break
      }
    }
  }

  for (let i = 0; i < routes.length; i++) {
    const dir = routes[i][0]
    const num = parseInt(routes[i][2])

    if (dir === 'N') {
      if (park[x - num]) {
        let obstacle = false
        for (let i = 1; i <= num; i++) {
          if (park[x - i][y] === 'X') {
            obstacle = true
            break
          }
        }
        if (obstacle) continue
        else x -= num
      } else continue
    } else if (dir === 'S') {
      if (park[x + num]) {
        let obstacle = false
        for (let i = 1; i <= num; i++) {
          if (park[x + i][y] === 'X') {
            obstacle = true
            break
          }
        }
        if (obstacle) continue
        else x += num
      } else continue
    } else if (dir === 'W') {
      if (park[x][y - num]) {
        let obstacle = false
        for (let i = 1; i <= num; i++) {
          if (park[x][y - i] === 'X') {
            obstacle = true
            break
          }
        }
        if (obstacle) continue
        else y -= num
      } else continue
    } else if (dir === 'E') {
      if (park[x][y + num]) {
        let obstacle = false
        for (let i = 1; i <= num; i++) {
          if (park[x][y + i] === 'X') {
            obstacle = true
            break
          }
        }
        if (obstacle) continue
        else y += num
      } else continue
    }
  }

  answer.push(x, y)

  return answer
}

function solution(today, terms, privacies) {
  var answer = []

  const todayArr = today.split('.')
  const todayYear = parseInt(todayArr[0])
  const todayMonth = parseInt(todayArr[1])
  const todayDay = parseInt(todayArr[2])

  const termsTable = {}
  for (let i = 0; i < terms.length; i++) {
    const termsArr = terms[i].split(' ')
    termsTable[termsArr[0]] = termsArr[1]
  }

  for (let i = 0; i < privacies.length; i++) {
    const privaciesArr = privacies[i].split(' ')
    const dateArr = privaciesArr[0].split('.')

    let expireYear = parseInt(dateArr[0])
    let expireMonth = parseInt(dateArr[1])
    let expireDay = parseInt(dateArr[2])

    const passMonth = parseInt(termsTable[privaciesArr[1]] % 12)
    const passYear = parseInt(termsTable[privaciesArr[1]] / 12)
    expireMonth += passMonth
    expireYear += passYear

    if (expireMonth > 12) {
      expireYear += 1
      expireMonth -= 12
    }

    expireDay -= 1
    if (expireDay === 0) {
      expireDay = 28
      expireMonth -= 1
      if (expireMonth === 0) {
        expireMonth = 12
        expireYear -= 1
      }
    }

    if (expireYear < todayYear) {
      answer.push(i + 1)
    } else if (expireYear === todayYear) {
      if (expireMonth < todayMonth) {
        answer.push(i + 1)
      } else if (expireMonth === todayMonth) {
        if (expireDay < todayDay) {
          answer.push(i + 1)
        }
      }
    }
  }
  return answer
}
```

function solution(number, limit, power) {
  var answer = 0

  for (let i = 1; i <= number; i++) {
    let count = 0

function solution(babbling) {

function solution(X, Y) {
    var answer = '';
    
    const table = {}
    const Xtable = {}
    const Ytable = {}
    
    for(let num of X) {
        if(Xtable[num]) Xtable[num]++;
        else Xtable[num] = 1;
    }
    for(let num of Y) {
        if(Ytable[num]) Ytable[num]++;
        else Ytable[num] = 1;
    }
    for(let num in Ytable) {
        if(Xtable[num]) {
            if(Xtable[num] <= Ytable[num]) table[num] = Xtable[num];
            else table[num] = Ytable[num];
        }
    }

    const keyArray = Object.keys(table); 
    if(keyArray.length === 0) return '-1';
    if(keyArray.length === 1 && keyArray[0] === '0') return '0';
    
    for(let num of keyArray.sort((a,b) => b-a)) {
        for(let i=0;i<table[num];i++) {
            answer += num;
        }
    }
    
    return answer;
}

function solution(survey, choices) {
    var answer = '';
    
    const table = {
        R:0,
        T:0,
        C:0,
        F:0,
        J:0,
        M:0,
        A:0,
        N:0
    }
    
    for(let i=0;i<survey.length;i++) {
        const choice = choices[i];
        if(choice === 1 || choice === 2 || choice === 3) {
            const firstChar = survey[i][0];
            
            if(choice === 1) table[firstChar] += 3;
            if(choice === 2) table[firstChar] += 2;
            if(choice === 3) table[firstChar] += 1;
        }
        else if(choice === 5 || choice === 6 || choice === 7) {
            const secondChar = survey[i][1];
            if(choice === 5) table[secondChar] += 1;
            if(choice === 6) table[secondChar] += 2;
            if(choice === 7) table[secondChar] += 3;
        }
    }
    
    if(table['R'] >= table['T']) answer += 'R';
    else answer += 'T';
    
    if(table['C'] >= table['F']) answer += 'C';
    else answer += 'F';
    
    if(table['J'] >= table['M']) answer += 'J';
    else answer += 'M';
    
    if(table['A'] >= table['N']) answer += 'A';
    else answer += 'N';
    
    return answer;
}
```

                }
            }
        }

            }

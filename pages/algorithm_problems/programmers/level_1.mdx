```js
function solution(players, callings) {
  var answer = []
  answer = Array.from(players)

  let temp_players = {}
  for (let i = 0; i < players.length; i++) {
    temp_players[players[i]] = i
  }

  for (let calling of callings) {
    let num = temp_players[calling]
    let temp = answer[num]
    answer[num] = answer[num - 1]
    answer[num - 1] = temp

    temp_players[answer[num]] = num
    temp_players[answer[num - 1]] = num - 1
  }

  return answer
}
```

## 추억 점수

```js
function solution(name, yearning, photo) {
  var answer = []

  let table = {}
  for (let i = 0; i < name.length; i++) {
    table[name[i]] = yearning[i]
  }

  for (let i = 0; i < photo.length; i++) {
    let score = 0
    for (let j = 0; j < photo[i].length; j++) {
      if (!table[photo[i][j]]) continue
      score += table[photo[i][j]]
    }
    answer.push(score)
  }

  return answer
}
```

## 공원 산책

https://school.programmers.co.kr/learn/courses/30/lessons/172928

```js
function solution(park, routes) {
  var answer = []

  let x, y

  for (let i = 0; i < park.length; i++) {
    for (let j = 0; j < park[i].length; j++) {
      if (park[i][j] === 'S') {
        x = i
        y = j
        break
      }
    }
  }

  for (let i = 0; i < routes.length; i++) {
    const dir = routes[i][0]
    const num = parseInt(routes[i][2])

    if (dir === 'N') {
      if (park[x - num]) {
        let obstacle = false
        for (let i = 1; i <= num; i++) {
          if (park[x - i][y] === 'X') {
            obstacle = true
            break
          }
        }
        if (obstacle) continue
        else x -= num
      } else continue
    } else if (dir === 'S') {
      if (park[x + num]) {
        let obstacle = false
        for (let i = 1; i <= num; i++) {
          if (park[x + i][y] === 'X') {
            obstacle = true
            break
          }
        }
        if (obstacle) continue
        else x += num
      } else continue
    } else if (dir === 'W') {
      if (park[x][y - num]) {
        let obstacle = false
        for (let i = 1; i <= num; i++) {
          if (park[x][y - i] === 'X') {
            obstacle = true
            break
          }
        }
        if (obstacle) continue
        else y -= num
      } else continue
    } else if (dir === 'E') {
      if (park[x][y + num]) {
        let obstacle = false
        for (let i = 1; i <= num; i++) {
          if (park[x][y + i] === 'X') {
            obstacle = true
            break
          }
        }
        if (obstacle) continue
        else y += num
      } else continue
    }
  }

  answer.push(x, y)

  return answer
}
```

  var answer = []

  let lux = (luy = 50)
  let rdx = (rdy = 0)

  for (let i = 0; i < wallpaper.length; i++) {
    for (let j = 0; j < wallpaper[i].length; j++) {
      if (wallpaper[i][j] === '#') {
        if (lux > i) lux = i
        if (luy > j) luy = j
        if (i > rdx) rdx = i
        if (j > rdy) rdy = j
      }
    }
  }

  return answer
}

function solution(n, m, section) {

  for (let i = 0; i < section.length; i++) {
